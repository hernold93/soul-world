var reader = "";
var current = 0;
var ratingSel = 0;
var KEY_PREFIX = "betafb_v2_";

function loadName(){ return localStorage.getItem(KEY_PREFIX+"name")||""; }
function storeName(n){ localStorage.setItem(KEY_PREFIX+"name",n); }
function fbKey(i){ return KEY_PREFIX+"scene_"+i; }
function loadFb(i){ 
  try { return JSON.parse(localStorage.getItem(fbKey(i))) || {}; } catch(e) { return {}; }
}
function saveFb(i, obj){ localStorage.setItem(fbKey(i), JSON.stringify(obj)); updateProgress(); }

function updateProgress(){
  let count = 0;
  for(let i=0; i<SCENES.length; i++){
    let fb = loadFb(i);
    if(fb.q1 || fb.rating) count++;
  }
  let pct = (count / SCENES.length) * 100;
  document.getElementById("progress-bar").style.width = pct + "%";
}

function go(name){
  document.querySelectorAll(".screen").forEach(s => s.classList.remove("active"));
  document.getElementById("s-"+name).classList.add("active");
  window.scrollTo(0,0);
}

function saveName(){
  var v = document.getElementById("reader-name").value.trim();
  if(!v) { alert("אנא הכנס שם"); return; }
  reader = v; storeName(v);
  document.getElementById("hello-name").textContent = "שלום, " + v;
  buildList(); go("list");
}

function buildList(){
  var c = document.getElementById("scene-list"); c.innerHTML = "";
  SCENES.forEach((s, i) => {
    let fb = loadFb(i);
    let isDone = fb.q1 || fb.rating;
    let card = document.createElement("div");
    card.className = "scene-card" + (isDone ? " completed" : "");
    card.onclick = () => openScene(i);
    card.innerHTML = `<div><strong>${i+1}. ${s.title}</strong></div>
                      <div style="font-size:0.8rem; color:var(--gold)">${isDone ? '✓ הושלם' : 'טרם נקרא'}</div>`;
    c.appendChild(card);
  });
  updateProgress();
}

function openScene(i){
  current = i; 
  let s = SCENES[i];
  let fb = loadFb(i);
  document.getElementById("read-crumb").textContent = s.title;
  
  let html = `<h2 class="display">${s.title}</h2><br>`;
  s.paras.forEach(p => html += `<p>${p}</p>`);
  
  html += `
    <div class="fb">
      <h3 class="display">משוב לסצנה</h3><br>
      <label>מה הכי תפס אותך?</label>
      <textarea id="f-q1" oninput="autoSave(${i})">${fb.q1||''}</textarea>
      <label>היה חלק פחות ברור / משעמם?</label>
      <textarea id="f-q2" oninput="autoSave(${i})">${fb.q2||''}</textarea>
      <label>הערות נוספות</label>
      <textarea id="f-q3" oninput="autoSave(${i})">${fb.q3||''}</textarea>
      <label>דירוג הסצנה</label>
      <div class="rating" id="rating"></div>
      <div class="center">
         <button class="btn" onclick="go('list')">סיימתי, חזרה לרשימה</button>
      </div>
    </div>`;
    
  document.getElementById("reader-body").innerHTML = html;
  renderStars(fb.rating || 0);
  go("read");
}

function renderStars(val){
  ratingSel = val;
  let container = document.getElementById("rating");
  container.innerHTML = "";
  for(let n=1; n<=5; n++){
    let span = document.createElement("span");
    span.className = "star" + (n <= val ? " on" : "");
    span.innerHTML = "★";
    span.onclick = () => {
      ratingSel = n;
      renderStars(n);
      autoSave(current);
    };
    container.appendChild(span);
  }
}

function autoSave(i){
  let obj = {
    q1: document.getElementById("f-q1").value,
    q2: document.getElementById("f-q2").value,
    q3: document.getElementById("f-q3").value,
    rating: ratingSel
  };
  saveFb(i, obj);
}

function generateFullReport(){
  let txt = `משוב בטא - "העולם שבנפש"\nמאת: ${reader}\n\n`;
  let any = false;
  SCENES.forEach((s, i) => {
    let fb = loadFb(i);
    if(fb.q1 || fb.rating) {
      any = true;
      txt += `--- ${s.title} ---\n`;
      if(fb.rating) txt += `דירוג: ${"★".repeat(fb.rating)}\n`;
      if(fb.q1) txt += `מה תפס: ${fb.q1}\n`;
      if(fb.q2) txt += `פחות ברור: ${fb.q2}\n`;
      if(fb.q3) txt += `הערות: ${fb.q3}\n\n`;
    }
  });
  return any ? txt : null;
}

function toast(m){
  let t = document.getElementById("toast");
  t.textContent = m; t.classList.add("show");
  setTimeout(() => t.classList.remove("show"), 3000);
}

// --- EVENTS ---
document.getElementById("btn-start").onclick = () => {
  if(loadName()) { reader=loadName(); buildList(); go("list"); }
  else go("intro");
};

document.getElementById("btn-savename").onclick = saveName;

document.getElementById("btn-export-wa").onclick = () => {
  let report = generateFullReport();
  if(!report) { toast("עדיין אין משוב לשלוח"); return; }
  window.open("https://wa.me/?text=" + encodeURIComponent(report));
};

document.getElementById("btn-copy").onclick = () => {
  let report = generateFullReport();
  if(!report) { toast("עדיין אין משוב להעתקה"); return; }
  navigator.clipboard.writeText(report).then(() => toast("המשוב הועתק ללוח!"));
};
